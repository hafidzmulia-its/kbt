% ---------- Problem ----------
\begin{frame}{Latar Belakang Masalah}{Mengapa ide ini relevan}
\begin{columns}[T,totalwidth=\textwidth]
\column{0.58\textwidth}
\begin{itemize}
  \item Banyak pasangan ingin undangan nikah digital yang \textbf{praktis}, tetapi tetap \textbf{terlihat personal dan elegan}.
  \item Banyak pengguna ingin membuat halaman spesial untuk hadiah, ulang tahun, wisuda, atau anniversary tanpa skill desain dan web.
  \item Solusi yang ada sering terpecah: ada yang kuat di template, ada yang kuat di tracking, tetapi belum sederhana bagi pengguna awam.
  \item Proses undang tamu masih sering manual dan sulit dipantau.
\end{itemize}

\column{0.38\textwidth}
\featurebox{Pain Point 1}{Sulit membuat undangan yang cepat namun tetap personal.}\\[0.18cm]
\featurebox{Pain Point 2}{Data tamu, RSVP, dan kehadiran sering tidak terkelola rapi.}\\[0.18cm]
\featurebox{Pain Point 3}{Gifting page yang menarik biasanya butuh effort desain tambahan.}
\end{columns}
\end{frame}

% ---------- Solution ----------
\begin{frame}{Konsep Solusi}{Undangan digital dan gifting site berbasis AI}
\begin{columns}[T,totalwidth=\textwidth]
\column{0.55\textwidth}
\begin{block}{Core value proposition}
NechCode membantu pengguna membuat undangan nikah digital dan gifting site berbasis template-AI dengan alur singkat: isi data, pilih template, hasil siap dibagikan, lalu RSVP dan kehadiran bisa dipantau.
\end{block}

\vspace{0.2cm}
\begin{itemize}
  \item Template visual dibantu AI
  \item Personalisasi nama tamu dan pesan
  \item Link undangan siap kirim ke WhatsApp
  \item RSVP tracking dan status hadir/tidak hadir
\end{itemize}
